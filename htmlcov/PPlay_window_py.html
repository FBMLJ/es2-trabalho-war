148#------------------------DRAW METHODS------------------------------- 

149 """ 

150 Draw a text on the screen at X and Y co-ords, using [R, G, B] color 

151 [with the specified font, 

152 [with the specified size, 

153 [Bold, 

154 [Italic]]]] 

155 """ 

156 def draw_text(self, text, x, y, size=12, color=(0,0,0), 

157 font_name="Arial", bold=False, italic=False): 

158 # Creates a Font from the system fonts 

159 # SysFont(name, size, bold=False, italic=False) -> Font 

160 font = pygame.font.SysFont(font_name, size, bold, italic) 

161 

162 # Creates a pygame.Surface with the text rendered on it 

163 # render(text, antialias, color, background=None)->Surface 

164 font_surface = font.render(text, True, color) 

165 # That's because pygame does NOT provide a way 

166 # to directly draw text on an existing Surface. 

167 # So you must use Font.render() -> Surface and BLIT 

168 

169 # Finally! BLIT! 

170 self.screen.blit(font_surface, [x, y])
