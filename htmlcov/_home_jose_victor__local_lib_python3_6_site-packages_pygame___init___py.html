42class MissingModule: 

43 _NOT_IMPLEMENTED_ = True 

44 

45 def __init__(self, name, urgent=0): 

46 self.name = name 

47 exc_type, exc_msg = sys.exc_info()[:2] 

48 self.info = str(exc_msg) 

49 self.reason = "%s: %s" % (exc_type.__name__, self.info) 

50 self.urgent = urgent 

51 if urgent: 

52 self.warn() 

53 

54 def __getattr__(self, var): 

55 if not self.urgent: 

56 self.warn() 

57 self.urgent = 1 

58 missing_msg = "%s module not available (%s)" % (self.name, self.reason) 

59 raise NotImplementedError(missing_msg) 

60 

61 def __nonzero__(self): 

62 return False 

63 

64 __bool__ = __nonzero__ 

65 

66 def warn(self): 

67 msg_type = 'import' if self.urgent else 'use' 

68 message = '%s %s: %s\n(%s)' % (msg_type, self.name, self.info, self.reason) 

69 try: 

70 import warnings 

71 level = 4 if self.urgent else 3 

72 warnings.warn(message, RuntimeWarning, level) 

73 except ImportError: 

74 print(message)

163def warn_unwanted_files(): 

164 """warn about unneeded old files""" 

165 

166 # a temporary hack to warn about camera.so and camera.pyd. 

167 install_path = os.path.split(pygame.base.__file__)[0] 

168 extension_ext = os.path.splitext(pygame.base.__file__)[1] 

169 

170 # here are the .so/.pyd files we need to ask to remove. 

171 ext_to_remove = ["camera"] 

172 

173 # here are the .py/.pyo/.pyc files we need to ask to remove. 

174 py_to_remove = ["color"] 

175 

176 # Don't warn on Symbian. The color.py is used as a wrapper. 

177 if os.name == "e32": 

178 py_to_remove = [] 

179 

180 # See if any of the files are there. 

181 extension_files = ["%s%s" % (x, extension_ext) for x in ext_to_remove] 

182 

183 py_files = ["%s%s" % (x, py_ext) 

184 for py_ext in [".py", ".pyc", ".pyo"] 

185 for x in py_to_remove] 

186 

187 files = py_files + extension_files 

188 

189 unwanted_files = [] 

190 for f in files: 

191 unwanted_files.append(os.path.join(install_path, f)) 

192 

193 ask_remove = [] 

194 for f in unwanted_files: 

195 if os.path.exists(f): 

196 ask_remove.append(f) 

197 

198 if ask_remove: 

199 message = "Detected old file(s). Please remove the old files:\n" 

200 

201 for f in ask_remove: 

202 message += "%s " % f 

203 message += "\nLeaving them there might break pygame. Cheers!\n\n" 

204 

205 try: 

206 import warnings 

207 level = 4 

208 warnings.warn(message, RuntimeWarning, level) 

209 except ImportError: 

210 print(message)
